function analyzeImage(){
  const file=document.getElementById("imageInput").files[0];
  if(!file) return alert("Please select an image file first.");

  const canvas=document.getElementById("canvas");
  const ctx=canvas.getContext("2d");

  function runAnalysis(){
    const issue=document.getElementById("issueType").value;
    document.getElementById("loader").style.display="block";

    setTimeout(()=>{
      document.getElementById("loader").style.display="none";

      const totalPx=canvas.width*canvas.height;
      const data=ctx.getImageData(0,0,canvas.width,canvas.height).data;

      /* ── DARK / LIGHT PIXEL ENGINE ──
         For every pixel compute luminance (perceptual weight).
         Bucket into: very dark (<50), dark (50–100), mid (100–155),
         light (155–210), very light (>210).                         */
      let veryDark=0, dark=0, mid=0, light=0, veryLight=0, totalLum=0;

      for(let i=0;i<data.length;i+=4){
        // Perceptual luminance (ITU-R BT.709)
        const lum=data[i]*0.2126 + data[i+1]*0.7152 + data[i+2]*0.0722;
        totalLum+=lum;
        if(lum<50)       veryDark++;
        else if(lum<100) dark++;
        else if(lum<155) mid++;
        else if(lum<210) light++;
        else             veryLight++;
      }

      const avgLum=totalLum/totalPx;
      const darkRatio=(veryDark+dark)/totalPx;      // 0–1
      const lightRatio=(light+veryLight)/totalPx;   // 0–1
      const veryDarkRatio=veryDark/totalPx;
      const veryLightRatio=veryLight/totalPx;
      const contrast=Math.abs(darkRatio-lightRatio); // higher = more varied surface

      /* ── Store pixel stats for results display ── */
      window._pixelStats={
        avgLum:avgLum.toFixed(1),
        darkPct:(darkRatio*100).toFixed(1),
        lightPct:(lightRatio*100).toFixed(1),
        veryDarkPct:(veryDarkRatio*100).toFixed(1),
        veryLightPct:(veryLightRatio*100).toFixed(1),
        contrast:(contrast*100).toFixed(1),
        totalPx
      };

      /* ── Minimum signal check: image must have some variation ── */
      if(darkRatio<0.01 && lightRatio<0.01){
        validDetection=false;
        document.getElementById("textResult").innerHTML=
          "<span style='color:#f87171'>⚠️ Image too uniform — upload a real road photo.</span>";
        openTab("results");
        return;
      }

      validDetection=true;

      /* ── Issue-specific dark/light logic ── */

      if(issue==="pothole"){
        /* Potholes = clusters of very dark pixels (shadows / depressions).
           More very-dark pixels → deeper/bigger pothole.              */
        const score=veryDarkRatio*100;               // 0–100
        const depth=(Math.sqrt(veryDarkRatio)*65).toFixed(1);
        lastIssue="Pothole";
        lastValue=`Depth est. ~${depth} cm  |  Dark pixel coverage: ${(veryDarkRatio*100).toFixed(1)}%`;
        lastSeverity=score<5?"Low":score<18?"Medium":"High";
        show(score*2);

      } else if(issue==="streetlight"){
        /* Working lights → bright hotspots (very light pixels).
           Many very-light pixels = lights on and working.
           Few very-light pixels = lights off / damaged.               */
        const score=veryLightRatio*100;
        lastIssue="Street Light";
        lastValue=`Bright pixel coverage: ${score.toFixed(1)}%  |  Avg luminance: ${avgLum.toFixed(0)}`;
        const status=score>15?"Operational":score>5?"Partial failure":"Not working / Off";
        lastSeverity=score>15?"Low":score>5?"Moderate":"Critical";
        window._pixelStats.extraNote=`Status: ${status}`;
        show(score>15?20:score>5?55:90);

      } else if(issue==="traffic"){
        /* Heavy traffic = many dark (vehicle) pixels against lighter road.
           High dark ratio + low average luminance = congested.        */
        const density=darkRatio*100;
        lastIssue="Traffic Density";
        lastValue=`Vehicle/dark coverage: ${density.toFixed(1)}%  |  Avg luminance: ${avgLum.toFixed(0)}`;
        lastSeverity=density<30?"Low":density<55?"Medium":"High";
        show(density);

      } else if(issue==="signal"){
        /* Signal lights appear as bright coloured spots.
           High very-light ratio + low average = lights visible.       */
        const hasSignal=veryLightRatio>0.02 || avgLum>100;
        lastIssue="Signal Issue";
        lastValue=`Bright spot ratio: ${(veryLightRatio*100).toFixed(1)}%  |  Avg lum: ${avgLum.toFixed(0)}`;
        lastSeverity=hasSignal?"Low":"High";
        window._pixelStats.extraNote=`Signal detected: ${hasSignal?"Yes ✅":"No ❌"}`;
        show(hasSignal?15:85);

      } else if(issue==="roadblock"){
        /* Blockages = large dark or very-dark mass in frame.
           High combined dark pixel ratio = obstructed road.           */
        const block=(veryDarkRatio+dark/totalPx)*100;
        lastIssue="Road Block";
        lastValue=`Obstruction coverage: ${block.toFixed(1)}%  |  Contrast: ${(contrast*100).toFixed(1)}%`;
        lastSeverity=block<20?"Low":block<45?"Medium":"High";
        show(block);
      }

      openTab("results");
    },1000);
  }

  // If canvas already has image data run directly, else load first
  if(canvas.width>0 && canvas.style.display!=="none"){
    runAnalysis();
  } else {
    const img=new Image();
    const url=URL.createObjectURL(file);
    img.src=url;
    img.onload=()=>{
      const maxW=860;
      let w=img.width, h=img.height;
      if(w>maxW){ h=Math.round(h*(maxW/w)); w=maxW; }
      canvas.width=w; canvas.height=h;
      canvas.style.display="block";
      canvas.style.marginTop="16px";
      ctx.drawImage(img,0,0,w,h);
      URL.revokeObjectURL(url);
      runAnalysis();
    };
    img.onerror=()=>alert("Could not load image.");
  }
}

function tmInit(){

    cell.className = 'cmap-cell';

    cmap.appendChild(cell);
  }

  // Build lane table rows
  const tbody = document.getElementById('lane-tbody');

  TM.lanes.forEach((lane, i)=>{

function tmTick(){
  if(TM.mode !== 'live') return;
  TM.tick++;

  // Countdown signals
  TM.signals.forEach((sig, i)=>{
    sig.countdown--;
    if(sig.countdown <= 0){
      // Cycle: green→yellow→red→green
      if(sig.state === 'green')       { sig.state = 'yellow'; sig.countdown = 4; }
      else if(sig.state === 'yellow') { sig.state = 'red';    sig.countdown = Math.round(20+Math.random()*20); }
      else                            { sig.state = 'green';  sig.countdown = Math.round(20+Math.random()*20); }
    }
  });

  // Fluctuate lane densities
  TM.lanes.forEach(lane=>{
    lane.density = Math.min(100, Math.max(5, lane.density + (Math.random()*10-5)));
  });

  // Fluctuate congestion map
  TM.cmap = TM.cmap.map(v=> Math.min(1, Math.max(0, v + (Math.random()*0.15-0.07))));

  // Update live stats
  const avgDensity = TM.lanes.reduce((a,l)=>a+l.density,0)/TM.lanes.length;
  TM.vehicles = Math.round(30 + avgDensity * 1.2 + Math.random()*8);
  TM.wait     = Math.round(10 + avgDensity * 0.8);
  TM.speed    = Math.round(Math.max(5, 60 - avgDensity * 0.55 + Math.random()*5));
  TM.incidents = TM.log.length;

  // Random incident ~ every 30s
  if(TM.tick % 30 === 0 && Math.random() > 0.4){
    tmAddIncident(INCIDENT_POOL[Math.floor(Math.random()*INCIDENT_POOL.length)]);
  }

  tmRenderAll();
}

function tmRenderAll(){
  // Stats
  document.getElementById('tm-vehicles').textContent = TM.vehicles;
  document.getElementById('tm-wait').textContent     = TM.wait + 's';
  document.getElementById('tm-speed').textContent    = TM.speed;
  document.getElementById('tm-incidents').textContent= TM.incidents;

  // Signals
  TM.signals.forEach((sig,i)=>{
    const light  = document.getElementById('sig-'+i+'-light');
    const status = document.getElementById('sig-'+i+'-status');
    const timer  = document.getElementById('sig-'+i+'-timer');
    light.className  = 'signal-light ' + sig.state;
    status.className = 'signal-status ' + sig.state;
    status.textContent = sig.state.toUpperCase();
    timer.textContent  = sig.countdown + 's';
  });

  // Lanes
  TM.lanes.forEach((lane,i)=>{
    const pct = Math.round(lane.density);
    const bar  = document.getElementById('lane-bar-'+i);
    const pctEl= document.getElementById('lane-pct-'+i);
    const badge= document.getElementById('lane-badge-'+i);

    let color = pct < 40 ? '#22c55e' : pct < 70 ? '#eab308' : '#ef4444';
    bar.style.width     = pct + '%';
    bar.style.background= color;
    pctEl.textContent   = pct + '%';

    if(pct < 40)       badge.innerHTML = '<span class="badge-low">Free</span>';
    else if(pct < 70)  badge.innerHTML = '<span class="badge-medium">Slow</span>';
    else               badge.innerHTML = '<span class="badge-high">Jammed</span>';
  });

  // Congestion map
  TM.cmap.forEach((v,i)=>{
    const cell = document.getElementById('cmap-'+i);
    if(!cell) return;
    if(v < 0.35)      cell.style.background = 'rgba(34,197,94,0.6)';
    else if(v < 0.65) cell.style.background = 'rgba(234,179,8,0.6)';
    else              cell.style.background = 'rgba(239,68,68,0.7)';
  });

  // Speed arc
  const maxSpeed = 60;
  const arc = document.getElementById('speed-arc');
  const gaugeVal = document.getElementById('gauge-val');
  const speedLabel = document.getElementById('speed-label');
  if(arc){
    const fraction = Math.min(TM.speed / maxSpeed, 1);
    arc.style.strokeDashoffset = 264 - fraction * 264;
    gaugeVal.textContent = TM.speed;
    speedLabel.textContent = TM.speed >= 40 ? '✅ Free Flow'
                           : TM.speed >= 20 ? '🟡 Slow Moving'
                           : '🔴 Heavy Jam';
  }
}

function tmAddIncident(incident){
  const now = new Date();
  const timeStr = now.getHours().toString().padStart(2,'0')+':'+
                  now.getMinutes().toString().padStart(2,'0')+':'+
                  now.getSeconds().toString().padStart(2,'0');
  TM.log.unshift({ ...incident, time: timeStr });
  if(TM.log.length > 8) TM.log.pop();

  // Re-render incident list
  const list = document.getElementById('incident-list');
  if(!list) return;
  list.innerHTML = TM.log.map(inc=>`
    <div class="incident-item">
      <div class="incident-dot ${inc.sev}"></div>
      <div class="incident-info">
        <div class="incident-title">${inc.title}</div>
        <div class="incident-meta">Severity: ${inc.sev.charAt(0).toUpperCase()+inc.sev.slice(1)}</div>
      </div>
      <div class="incident-time">${inc.time}</div>
    </div>`).join('');
}

  });
  document.getElementById("loginMsg").textContent="";
  document.getElementById("signupMsg").textContent="";
}

function setMsg(id, text, type){
  const el=document.getElementById(id);
  el.textContent=text;
  el.className="auth-msg "+(type||"");
}

function doSignup(){
  const name=document.getElementById("signupName").value.trim();
  const email=document.getElementById("signupEmail").value.trim().toLowerCase();
  const pass=document.getElementById("signupPass").value;
  const pass2=document.getElementById("signupPass2").value;

  if(!name||!email||!pass||!pass2) return setMsg("signupMsg","All fields are required.","error");
  if(!/\S+@\S+\.\S+/.test(email)) return setMsg("signupMsg","Enter a valid email.","error");
  if(pass.length<6) return setMsg("signupMsg","Password must be at least 6 characters.","error");
  if(pass!==pass2) return setMsg("signupMsg","Passwords do not match.","error");

  const users=getUsers();
  if(users[email]) return setMsg("signupMsg","Account already exists. Please login.","error");

  users[email]={name, pass};
  saveUsers(users);
  setMsg("signupMsg","Account created! Switching to login…","success");
  setTimeout(()=>switchAuth("login"),1200);
}

function doLogin(){
  const email=document.getElementById("loginEmail").value.trim().toLowerCase();
  const pass=document.getElementById("loginPass").value;

  if(!email||!pass) return setMsg("loginMsg","Please enter email and password.","error");

  const users=getUsers();
  if(!users[email]||users[email].pass!==pass) return setMsg("loginMsg","Invalid email or password.","error");

  // Success — hide overlay, show user chip
  document.getElementById("authOverlay").style.display="none";
  const chip=document.getElementById("userChip");
  chip.textContent="👤 "+users[email].name;
  chip.style.display="inline";
  document.getElementById("logoutBtn").style.display="inline";
}

function doLogout(){
  document.getElementById("authOverlay").style.display="flex";
  document.getElementById("userChip").style.display="none";
  document.getElementById("logoutBtn").style.display="none";
  document.getElementById("loginEmail").value="";
  document.getElementById("loginPass").value="";
  setMsg("loginMsg","","");
  switchAuth("login");
}
